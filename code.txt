# 网络爬取的充电站信息
highway_station_web = pd.read_csv(r'newdata/chargestation_guangdong.csv')
highway_station_web = gpd.GeoDataFrame(highway_station_web)
highway_station_web['geometry']  = gpd.points_from_xy(highway_station_web['lon'],highway_station_web['lat'])

highway_station = highway_station_web.iloc[:100][['geometry']]

# 判断是否为已有的充电站
highway_station_web.crs = 'epsg:4326'
highway_station.crs = 'epsg:4326'
nearest_highway_station = tbd.ckdnearest_point(highway_station_web.to_crs('epsg:4525'),highway_station.to_crs('epsg:4525'))

# 额外的充电站信息
extra_highway_station = nearest_highway_station[nearest_highway_station['dist']>500][['lon','lat','场站名称']]

extra_highway_station['id'] = range(len(extra_highway_station))
extra_highway_station['id'] += max(highway_station['id'])+1
